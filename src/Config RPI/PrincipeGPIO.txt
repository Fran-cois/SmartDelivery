
ETAPE 1:
sudo -i // entrer en SuperUser(root)


ETAPE 2: export gpios
root@raspberrypi:~# echo string1 > /sys/class/gpio/export
root@raspberrypi:~# echo string2 > /sys/class/gpio/export // Exporter les GPIOs à utiliser


ETAPE 3: set direction
root@raspberrypi:~# echo "out" > /sys/class/gpio/gpiostring1/direction
root@raspberrypi:~# echo "in" > /sys/class/gpio/gpiostring2/direction // Régler les directions des GPIOs


ETAPE4: set value
root@raspberrypi:~# echo "1" > /sys/class/gpio/gpiostring1/value // On change la valeur du GPIO en 1 pour qu'il transmette le courant en 3.3V

=====> LED ALLUMEE

ETAPE5: unexport gpios
root@raspberrypi:~#echo string1 > /sys/class/gpio/unexport
root@raspberrypi:~#echo string2 > /sys/class/gpio/unexport
